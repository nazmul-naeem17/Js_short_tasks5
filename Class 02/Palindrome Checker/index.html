Make a Palindrome Checker function

function reverse(str) { return str.split('').reverse().join(''); }function isPalindrome(...strings) { strings.forEach(str => { let newStr = reverse(str); if (str.toLowerCase() === newStr.toLowerCase()) { console.log("${str}" is a Palindrome word.); } else { console.log("${str}" is not a Palindrome word.); } }); }
